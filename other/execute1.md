Using a while loop to iterate over an array and push its elements into a new array:
css
Copy code
var arr1 = [1, 2, 3, 4, 5];
var arr2 = [];
var i = 0;
while (i < arr1.length) {
  arr2.push(arr1[i]);
  i++;
}
return arr2;
